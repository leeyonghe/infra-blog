                    }

                    steps {
                        sh 'npm audit'
                        sh 'docker run --rm -v $(pwd):/app clair-scanner --ip $(hostname -i) myapp:latest'

                }
            }
        }
        
        stage('Build Docker Image') {
            steps {
                script {
                    def image = docker.build("${DOCKER_REGISTRY}/${APP_NAME}:${env.GIT_COMMIT_SHORT}")

                        image.push()
